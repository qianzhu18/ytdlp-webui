@dataclass
class Job:
    job_id: str
    url: str
    preset: str
    use_cookies: bool
    created_at: float = field(default_factory=time.time)
    title: str = "Fetching info..."
    status: str = "Queued"

    error: str | None = None

def progress_hook(job):
    def hook(d):
        if d['status'] == 'downloading':
            total = d.get('total_bytes') or d.get('total_bytes_estimate')
            downloaded = d.get('downloaded_bytes')
            with job.lock:
                if total:
                    job.progress = (downloaded / total) * 100
                job.status = d.get('_percent_str', '0%') + ' ' + d.get('_speed_str', '')
                # å°è¯•æŠ“å–æ ‡é¢˜
                if d.get('info_dict'):
                    job.title = d['info_dict'].get('title', job.title)
        elif d['status'] == 'finished':
            with job.lock:
                job.progress = 100
                job.status = "Processing..."
    return hook

def worker(job_id):
    job = jobs.get(job_id)
    if not job: return
    
    with job.lock:
        job.status = "Starting..."

    opts = {
        "format": FORMAT_PRESETS[job.preset]["format"],
        "outtmpl": str(Path(DOWNLOAD_DIR) / "%(title)s.%(ext)s"),
        "progress_hooks": [progress_hook(job)],
        "quiet": True,
        "no_warnings": True,
        "ignoreerrors": True
    }
    
    preset_conf = FORMAT_PRESETS[job.preset]
    if "merge_output_format" in preset_conf:
        opts["merge_output_format"] = preset_conf["merge_output_format"]
    if "postprocessors" in preset_conf:
        opts["postprocessors"] = preset_conf["postprocessors"]
    if job.use_cookies and COOKIES_PATH:
        opts["cookiefile"] = COOKIES_PATH

    try:
        with yt_dlp.YoutubeDL(opts) as ydl:
            # å…ˆå¿«é€Ÿè·å–æ ‡é¢˜ï¼ˆæå‡ä½“éªŒï¼‰
            try:
                info = ydl.extract_info(job.url, download=False)
                with job.lock:
                    job.title = info.get('title', job.url)
            except: pass
            
            ydl.download([job.url])
            
        with job.lock:
            job.done = True
            job.status = "Done"
            job.progress = 100
    except Exception as e:
        with job.lock:
            job.done = True
            job.status = "Failed"
            job.error = str(e)

@app.route("/api/start", methods=["POST"])
def start():
    data = request.json or {}
    urls = [u.strip() for u in data.get("url", "").splitlines() if u.strip()]
    if not urls: return jsonify({"error": "No URL"}), 400
    
    added = 0
    with jobs_lock:
        for u in urls:
            jid = uuid.uuid4().hex
            job = Job(jid, u, data.get("preset"), data.get("use_cookies"))
            jobs[jid] = job
            executor.submit(worker, jid)
            added += 1
    return jsonify({"count": added})

@app.route("/api/tasks")
def tasks():
    with jobs_lock:
        # åªè¿”å›æœ€è¿‘20æ¡ï¼ŒæŒ‰æ—¶é—´å€’åº
        ret = []
        for j in sorted(jobs.values(), key=lambda x: x.created_at, reverse=True)[:20]:
            ret.append({
                "id": j.job_id, "title": j.title, "status": j.status,
                "progress": round(j.progress), "done": j.done, "error": j.error
            })
    return jsonify({"tasks": ret})
